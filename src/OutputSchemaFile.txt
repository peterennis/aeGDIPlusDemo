Public Sub CreateTheDb()

Dim strSQL As String
On Error GoTo ErrorTrap

'Local Table

strSQL="CREATE TABLE [tblImages] ([ID] Counter,[Image] Attachment )"

strSQL="CREATE INDEX [ID] ON [tblImages] ([Image]) "
Currentdb.Execute strSQL

strSQL="CREATE UNIQUE INDEX [PrimaryKey] ON [tblImages] ([Image])  WITH PRIMARY DISALLOW NULL "
Currentdb.Execute strSQL
